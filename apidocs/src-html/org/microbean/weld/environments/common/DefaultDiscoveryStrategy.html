017package org.microbean.weld.environments.common;
018
019import java.lang.annotation.Annotation;
020
021import java.security.AccessController;
022import java.security.PrivilegedAction;
023
024import java.util.Arrays;
025import java.util.Set;
026
027import org.jboss.weld.bootstrap.api.Bootstrap;
028
029import org.jboss.weld.environment.logging.CommonLogger;
030
031import org.jboss.weld.environment.deployment.WeldBeanDeploymentArchive;
032
033import org.jboss.weld.environment.deployment.discovery.BeanArchiveScanner;
034import org.jboss.weld.environment.deployment.discovery.DiscoveryStrategy;
035import org.jboss.weld.environment.deployment.discovery.DiscoveryStrategyFactory; // for javadoc only
036import org.jboss.weld.environment.deployment.discovery.ReflectionDiscoveryStrategy;
037
038import org.jboss.weld.environment.deployment.discovery.jandex.Jandex;
039
040import org.jboss.weld.resources.spi.ResourceLoader;
041
042/**
043 * A {@link DelegatingDiscoveryStrategy} that delegates to a {@link
044 * DiscoveryStrategy} that is virtually 100% compatible with the
045 * default Weld behavior.
046 *
047 * <p>This class exists primarily for those cases where end users wish
048 * to provide their own {@link DiscoveryStrategy} via the Java SPI
049 * mechanism, but need to fall back to default behavior, whatever that
050 * may be, if something is not right.</p>
051 *
052 * <p>This class has been designed like other {@link
053 * DiscoveryStrategy} implementations to be instantiated by the {@link
054 * DiscoveryStrategyFactory} class.  Normally that means its instances
055 * are found in a Java SPI {@code META-INF/services} service provider
056 * file.  Normally a {@link DiscoveryStrategy} implementation is
057 * instantiated, and then its {@link
058 * #setResourceLoader(ResourceLoader)}, {@link
059 * #setBootstrap(Bootstrap)}, {@link
060 * #setInitialBeanDefiningAnnotations(Set)} and {@link
061 * #setScanner(BeanArchiveScanner)} methods are called.  Subclasses or
062 * callers should be prepared for any fields that might be set by
063 * these calls to be {@code null} at {@link #performDiscovery()} time.</p>
064 *
065 * <h2>Thread Safety</h2>
066 *
067 * <p>Instances of this class are not safe for concurrent use by
068 * multiple threads.</p>
069 *
070 * @author <a href="https://about.me/lairdnelson"
071 * target="_parent">Laird Nelson</a>
072 *
073 * @see DelegatingDiscoveryStrategy
074 *
075 * @see DiscoveryStrategyFactory
076 */
077public class DefaultDiscoveryStrategy extends DelegatingDiscoveryStrategy {
078
079  protected ResourceLoader resourceLoader;
080
081  protected Bootstrap bootstrap;
082
083  protected Set<Class<? extends Annotation>> initialBeanDefiningAnnotations;
084
085  protected BeanArchiveScanner beanArchiveScanner;
086
087  public DefaultDiscoveryStrategy() {
088    this(null);
089  }
090
091  public DefaultDiscoveryStrategy(final DiscoveryStrategy delegate) {
092    super(delegate);
093  }
094
095  @Override
096  public void setResourceLoader(final ResourceLoader resourceLoader) {
097    super.setResourceLoader(resourceLoader);
098    this.resourceLoader = resourceLoader;
099  }
100
101  @Override
102  public void setBootstrap(final Bootstrap bootstrap) {
103    super.setBootstrap(bootstrap);
104    this.bootstrap = bootstrap;
105  }
106
107  @Override
108  public void setInitialBeanDefiningAnnotations(final Set<Class<? extends Annotation>> initialBeanDefiningAnnotations) {
109    super.setInitialBeanDefiningAnnotations(initialBeanDefiningAnnotations);
110    this.initialBeanDefiningAnnotations = initialBeanDefiningAnnotations;
111  }
112
113  @Override
114  public void setScanner(final BeanArchiveScanner beanArchiveScanner) {
115    super.setScanner(beanArchiveScanner);
116    if (this.beanArchiveScanner == null) {
117      this.beanArchiveScanner = beanArchiveScanner;
118    }
119  }
120
121  @Override
122  public Set<WeldBeanDeploymentArchive> performDiscovery() {
123    this.delegate = acquireDelegate();
124    return super.performDiscovery();
125  }
126
127  protected DiscoveryStrategy acquireDelegate() {
128    DiscoveryStrategy returnValue = null;
129    if (this.delegate == null) {
130      if (Jandex.isJandexAvailable(this.resourceLoader)) {
131        if (isJandexStrategyDisabled()) {
132          CommonLogger.LOG.jandexDiscoveryStrategyDisabled();
133        } else {
134          CommonLogger.LOG.usingJandex();
135          try {
136            returnValue = Jandex.createJandexDiscoveryStrategy(this.resourceLoader, this.bootstrap, this.initialBeanDefiningAnnotations);
137          } catch (final Exception exception) {
138            throw CommonLogger.LOG.unableToInstantiate(Jandex.JANDEX_DISCOVERY_STRATEGY_CLASS_NAME,
139                                                       Arrays.toString(new Object[] { this.resourceLoader,
140                                                                                      this.bootstrap,
141                                                                                      this.initialBeanDefiningAnnotations }),
142                                                       exception);
143          }
144        }
145      }
146      if (returnValue == null) {
147        returnValue = new ReflectionDiscoveryStrategy(this.resourceLoader, this.bootstrap, this.initialBeanDefiningAnnotations);
148      }
149      assert returnValue != null;
150      if (this.beanArchiveScanner != null) {
151        returnValue.setScanner(this.beanArchiveScanner);
152      }
153      this.delegateAcquired(returnValue);
154    } else {
155      returnValue = this.delegate;
156    }
157    return returnValue;
158  }
159
160  protected void delegateAcquired(final DiscoveryStrategy delegate) {
161
162  }
163
164  private static final boolean isJandexStrategyDisabled() {
165    return (Boolean)AccessController.doPrivileged((PrivilegedAction)() -> Boolean.getBoolean(Jandex.DISABLE_JANDEX_DISCOVERY_STRATEGY));
166  }
167
168}
